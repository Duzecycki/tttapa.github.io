 145     switch (formatPrintStream)

 191 static char nibble_to_hex(uint8_t nibble) {  // convert a 4-bit nibble to a hexadecimal character
 192   nibble &= 0xF;
 193   return nibble > 9 ? nibble - 10 + ('a' & casePrintStream) : nibble + '0';
 194 }

 200 template <class T>
 201 void printHex(Print &printer, T val)
 202 {
 203     if (showbasePrintStream)
 204         printer.print(F("0x"));
 205     bool nonZero = false;
 206     for (int i = sizeof(val) - 1; i >= 0; i--)
 207     {
 208         uint8_t currByte = ((uint8_t *)&val)[i];
 209         if (currByte != 0 || i == 0)
 210             nonZero = true;
 211         if (leadingZerosPrintStream || nonZero) {
 212             printer.print(nibble_to_hex(currByte >> 4));
 213             printer.print(nibble_to_hex(currByte));
 214             if (byteSeparatorPrintStream && i)
 215                 printer.print(byteSeparatorPrintStream);
 216         }
 217     }
 218 }

 220 template <class T>
 221 void printBin(Print &printer, T val)
 222 {
 223     if (showbasePrintStream)
 224         printer.print(F("0b"));
 225     bool nonZero = false;
 226     for (int i = sizeof(val) - 1; i >= 0; i--)
 227     {
 228         uint8_t currByte = ((uint8_t *)&val)[i]; 
 229         for (int j = 7; j >= 0; j--)
 230         {
 231             uint8_t currBit = currByte & 0x80;
 232             if (currBit != 0 || (i == 0 && j == 0))
 233                 nonZero = true;
 234             if (leadingZerosPrintStream || nonZero)
 235                 printer.print(currBit ? '1' : '0');
 236             currByte <<= 1;
 237         }
 238         if (byteSeparatorPrintStream && i && (leadingZerosPrintStream || nonZero))
 239             printer.print(byteSeparatorPrintStream);
 240     }
 241 }
