  34      * When the switch is in the 'off' position, Setting 1 is selected
  35      * When the switch is in the 'on' position, Setting 2 is selected

 111     Selector(pin_t switchPin, pin_t ledPin,

 125      * @brief Construct a Selector object with multiple buttons.
 126 
 127      * <h3>Multiple momentary switches (push buttons)</h3>
 128      *
 129      * Pressing one of the buttons selects the respective output:
 130      * When starting the program, Setting 1 is selected,
 131      * When the second button is pressed, Setting 2 is selected,
 132      * When the N-th button is pressed, Setting N is selected.
 133      *
 134      * Selector( { buttonPin1, buttonPin2, ... , buttonPinN } );

 136     @image html "selector-multiple-momentary-switches.svg"

 143     template <size_t N>
 144     Selector(const pin_t (&switchPins)[N]) // Multiple buttons, no LEDs
 145         : nb_settings(N) {
 146         static_assert(N > 1, "Error: minimum number of switches is 2. ");
 147         this->switchPins = new pin_t[N];
 148         copy(this->switchPins, switchPins);
 149         mode = MULTIPLE_BUTTONS;
 150         for (uint8_t i = 0; i < nb_settings; i++)
 151             pinMode(this->switchPins[i], INPUT_PULLUP);
 152         LinkedList::append(this, first, last);

 154     /**
 155      * @brief   Construct a Selector object with multiple buttons and
 156      *          multiple LEDs.
 157 
 158      * <h3>Multiple momentary switches (push buttons) and multiple LEDs</h3>
 159      *
 160      * Pressing one of the buttons selects the respective output and
 161      * enables the respective LED:
 162      * When starting the program, Setting 1 is selected and LED 1 is on,
 163      * When the second button is pressed, Setting 2 is selected, LED 1 turns off
 164      * and LED 2 turns on,
 165      * When the N-th button is pressed, Setting N is selected, the N-th LED
 166      * turns on, and all other LEDs are off.
 167      *
 168      * Selector( { buttonPin1, buttonPin2, ... , buttonPinN },
 169      * { ledPin1, ledPin2, ... , ledPinN } );
 170 
 171     @image html "selector-multiple-momentary-switches-LED.svg"
 172 
 173      * <h3>Two momentary switches (push buttons) and multiple LEDs
 174      * (increment / decrement)</h3>
 175      *
 176      * Pressing the first button increments the Setting number and turns on the
 177      * respective LED,
 178      * pressing the second button decrements the Setting number and turns on the
 179      * respective LED:
 180      * When starting the program, Setting 1 is selected and LED 1 is on,
 181      * When the first button is pressed, Setting 2 is selected, LED 1 turns off
 182      * and LED 2 turns on,
 183      * When the first button is pressed again, Setting 3 is selected, LED 2
 184      * turns off and LED 3 turns on.
 185      * When the last Setting is selected, and the first button is pressed,
 186      * Setting 1 is selected, the last LED turns off and LED 1 turns on.
 187      * When the second button is pressed, the last Setting (M) is selected,
 188      * LED 1 turns off and LED M turns on,
 189      * When the second button is pressed again, Setting (M-1) is selected,
 190      * LED M turns off and LED M-1 turns on,
 191      * and so on.
 192      *
 193      * Selector( { incrementButtonPin, decrementButtonPin },
 194      * { ledPin1, ledPin2, ... , ledPinM });
 195 
 196     @image html "selector-increment-decrement-LED.svg"
 197 
 198      * <h3>One momentary switch (push button) and multiple LEDs (increment)</h3>
 199      *
 200      * Pressing the button increments the Setting number and turns on the
 201      * respective LED,
 202      * When starting the program, Setting 1 is selected and LED 1 is on,
 203      * When the button is pressed, Setting 2 is selected, LED 1 turns off and
 204      * LED 2 turns on,
 205      * When the button is pressed again, Setting 3 is selected, LED 2 turns off
 206      * and LED 3 turns on.
 207      * When the M-th Setting is selected, and the button is pressed,
 208      * Setting 1 is selected, the M-th LED turns off and LED 1 turns on.
 209      *
 210      * Selector( { incrementButtonPin }, { ledPin1, ledPin2, ... , ledPinM });
 211 
 212     @image html "selector-increment-LED.svg"
 213 
 214      * @tparam  N
 215      *          The number of buttons.
 216      * @tparam  M
 217      *          The number of LEDs.
 218      * @param   switchPins
 219      *          A list of pin numbers with the buttons connected.
 220      * @param   ledPins
 221      *          A list of pin numbers with the LEDs connected.
 222      */
 223     template <size_t N, size_t M>
 224     Selector(
 225         const pin_t (&switchPins)[N],
 226         const pin_t (&ledPins)[M]) // One or multiple buttons, multiple LEDs
 227         : nb_settings(M) {
 228         // TODO: can this ever happen?
 229         static_assert(N > 0, "Error: no switch pins specified. ");
 230         // TODO: can this ever happen?
 231         static_assert(M > 0, "Error: no LED pins specified. ");
 232         static_assert(N <= 2 || N == M, "Error: number of LEDs should be the "
 233                                         "same as the number of buttons. ");
 234 
 235         this->ledPins = new pin_t[M];
 236         copy(this->ledPins, ledPins);
 237         if (N == 1) {
 238             dbButton1.pin = switchPins[0];
 239             mode = INCREMENT_LEDS;
 240             pinMode(dbButton1.pin, INPUT_PULLUP);
 241         } else if (N == 2) {
 242             dbButton1.pin = switchPins[0];
 243             dbButton2.pin = switchPins[1];
 244             mode = INCREMENT_DECREMENT_LEDS;
 245             pinMode(dbButton1.pin, INPUT_PULLUP);
 246             pinMode(dbButton2.pin, INPUT_PULLUP);
 247         } else {
 248             this->switchPins = new pin_t[N];
 249             copy(this->switchPins, switchPins);
 250             mode = MULTIPLE_BUTTONS_LEDS;
 251             for (uint8_t i = 0; i < nb_settings; i++)
 252                 pinMode(switchPins[i], INPUT_PULLUP);
 253         }
 254         initLEDs();
 255         LinkedList::append(this, first, last);
 256     }

 258     /**
 259      * @brief Construct a Selector object with increment / decrement buttons.
 260 
 261      * <h3>Two momentary switches (push buttons) (increment / decrement)</h3>
 262      *
 263      * Pressing the first button increments the Setting number,
 264      * pressing the second button decrements the Setting number:
 265      * When starting the program, Setting 1 is selected,
 266      * When the first button is pressed, Setting 2 is selected,
 267      * When the first button is pressed again, Setting 3 is selected,
 268      * When the last Setting is selected, and the first button is pressed again,
 269      * Setting 1 is selected.
 270      * When the second button is pressed, the last Setting (nb_settings) is
 271      * selected,
 272      * When the second button is pressed again, the previous Setting
 273      * (nb_settings-1) is selected,
 274      * and so on.
 275      *
 276      * Selector ( { incrementButtonPin, decrementButtonPin }, nb_settings );
 277 
 278     @image html "selector-increment-decrement.svg"
 279 
 280      * <h3>One momentary switch (push button) (increment)</h3>
 281      *
 282      * Pressing the button increments the Setting number,
 283      * When starting the program, Setting 1 is selected,
 284      * When the button is pressed, Setting 2 is selected,
 285      * When the button is pressed again, Setting 3 is selected,
 286      * When the last Setting is selected, and the button is pressed again,
 287      * Setting 1 is selected.
 288      *
 289      * Selector( { incrementButtonPin }, nb_settings );
 290 
 291     @image html "selector-increment.svg"
 292 
 293      * @tparam  N
 294      *          The number of buttons [1, 2].
 295      * @param   switchPins
 296      *          A list of pin numbers with the buttons connected.
 297      * @param   nb_settings
 298      *          The number of settings to cycle through.
 299      */
 300     template <size_t N>
 301     Selector(const pin_t (&switchPins)[N],
 302              uint8_t nb_settings) // Two buttons (+1, -1), no LEDs
 303         : nb_settings(nb_settings) {
 304         // TODO: can this ever happen?
 305         static_assert(N > 0, "Error: no switch pins specified. ");
 306         static_assert(N <= 2, "Error: maximum number of buttons in the "
 307                               "increment/decrement configuration is 2. ");
 308         if (N == 1) {
 309             dbButton1.pin = switchPins[0];
 310             mode = INCREMENT;
 311             pinMode(dbButton1.pin, INPUT_PULLUP);
 312             LinkedList::append(this, first, last);
 313         } else if (N == 2) {
 314             dbButton1.pin = switchPins[0];
 315             dbButton2.pin = switchPins[1];
 316             mode = INCREMENT_DECREMENT;
 317             pinMode(dbButton1.pin, INPUT_PULLUP);
 318             pinMode(dbButton2.pin, INPUT_PULLUP);
 319             LinkedList::append(this, first, last);
 320         }
 321     }
