  10 /**
  11  * @brief   A class for MIDI interfaces sending and receiving 
  12  *          human-readable MIDI messages over a Stream.
  13  */
  14 class StreamDebugMIDI_Interface : public StreamMIDI_Interface {
  15   public:
  16     /**
  17      * @brief   Construct a debug MIDI interface on the given Stream.
  18      * 
  19      * @param   stream
  20      *          The Stream interface.
  21      */
  22     StreamDebugMIDI_Interface(Stream &stream) : StreamMIDI_Interface(stream) {}
  23 
  24   protected:
  25     void sendImpl(uint8_t m, uint8_t c, uint8_t d1, uint8_t d2) override {
  26         uint8_t messageType = (m >> 4) - 8;
  27         if (messageType >= 7)
  28             return;
  29         stream.print(MIDI_STATUS_TYPE_NAMES[messageType]);
  30         stream.print("\tChannel: ");
  31         stream.print(c + 1);
  32         stream.print("\tData 1: 0x");
  33         stream.print(d1, HEX);
  34         stream.print("\tData 2: 0x");
  35         stream.print(d2, HEX);
  36         stream.print("\r\n");
  37         stream.flush();
  38     }
  39 
  40     void sendImpl(uint8_t m, uint8_t c, uint8_t d1) override {
  41         uint8_t messageType = (m >> 4) - 8;
  42         if (messageType >= 7)
  43             return;
  44         stream.print(MIDI_STATUS_TYPE_NAMES[messageType]);
  45         stream.print("\tChannel: ");
  46         stream.print(c + 1);
  47         stream.print("\tData 1: 0x");
  48         stream.print(d1, HEX);
  49         stream.print("\r\n");
  50         stream.flush();
  51     }
  52 
  53     virtual MIDI_read_t read() override {
  54         while (stream.available() > 0) {
  55             char data = stream.read();
  56 
  57             if (isHexChar(toLowerCase(data))) {
  58                 data = toLowerCase(data);
  59                 if (firstChar == '\0') {
  60                     firstChar = data;
  61                 } else if (secondChar == '\0') {
  62                     secondChar = data;
  63                 } else {
  64                     firstChar = secondChar;
  65                     secondChar = data;
  66                 }
  67             } else if (isWhiteSpace(data) && firstChar && secondChar) {
  68                 // if we received two hex characters followed by
  69                 // whitespace
  70                 uint8_t midiByte = hexCharToNibble(firstChar) << 4 |
  71                                    hexCharToNibble(secondChar);
  72                 firstChar = '\0';
  73                 secondChar = '\0';
  74                 MIDI_read_t parseResult = parser.parse(midiByte);
  75                 if (parseResult != NO_MESSAGE)
  76                     return parseResult;
  77             } else {
  78                 ; // Ignore any characters other than whitespace and hexadecimal digits
  79             }
  80         }
  81         return NO_MESSAGE;
  82     }
  83 
  84   private:
  85     char firstChar;
  86     char secondChar;
  87 
  88     /**
  89      * @brief   Check if a given character is a lowercase hexadecimal digit
  90      *          (0-9 or a-f).
  91      * 
  92      * @param   hex
  93      *          The hexadecimal character to check.
  94      * @return  true
  95      *          The character is a lowercase hexadecimal digit.
  96      * @return  false
  97      *          The character is not a lowercase hexadecimal digit.
  98      */
  99     bool isHexChar(char hex) {
 100         return (hex >= '0' && hex <= '9') || (hex >= 'a' && hex <= 'f');
 101     }
 102     /**
 103      * @brief   Convert a given letter to lower case, no effect if
 104      *          it's a digit (0 - 9) or if it's already lower case.
 105      */
 106     char toLowerCase(char x) { return x | 0b0100000; }
 107     /**
 108      * @brief   Convert a hexadecimal character to a 4-bit nibble.
 109      */
 110     uint8_t hexCharToNibble(char hex) {
 111         return hex < 'a' ? hex - '0' : hex - 'a' + 10;
 112     }
 113     /**
 114      * @brief   Check if the given character is whitespace (SP, CR or LF).
 115      * 
 116      * @return  true
 117      *          The given character is either a space,
 118      *          a carriage return or a line feed (SP, CR or LF).
 119      */
 120     inline bool isWhiteSpace(char x) {
 121         return x == ' ' || x == '\r' || x == '\n';
 122     }
 123 };

 125 /**
 126  * @brief   A wrapper class for debug MIDI interfaces sending and receiving 
 127  *          human-readable MIDI messages over a Serial port of class T.
 128  * 
 129  * @note    This is a template class because the class of the Serial object
 130  *          is completely different on different architectures, and they
 131  *          do not share a common super-class that has a `begin` method.
 132  */
 133 template <typename T>
 134 class SerialDebugMIDI_Interface : public StreamDebugMIDI_Interface {
 135   public:
 136     /**
 137      * @brief   Construct a new Debug MIDI Interface on the given Serial interface
 138      *          with the given baud rate.
 139      * 
 140      * @param   serial
 141      *          The Serial interface.
 142      * @param   baud
 143      *          The baud rate for the Serial interface.
 144      */
 145     SerialDebugMIDI_Interface(T &serial, unsigned long baud)
 146         : StreamDebugMIDI_Interface(serial), serial(serial), baud(baud) {}
 147     /**
 148      * @brief   Start the Serial interface at the predefined baud rate.
 149      */
 150     virtual void begin() override { serial.begin(baud); }
 151 
 152   private:
 153     T &serial;
 154     const unsigned long baud;
 155 };
