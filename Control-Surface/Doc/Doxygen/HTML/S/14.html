 107 template <size_t NUMBER_OF_BANKS>
 108 class VU : public virtual VU_Base, public BankableMIDIInputAddressable {
 109   public:
 110     VU(uint8_t track, uint8_t channel = 1, unsigned int decayTime = 300)
 111         : VU_Base(track, channel, decayTime) {}
 112 
 113     bool updateImpl(const MIDI_message_matcher &midimsg) {
 114         uint8_t targetTrack = midimsg.data1 >> 4;
 115         DEBUGFN("target track = " << +targetTrack);
 116         if (!matchTrack(targetTrack))
 117             return false;
 118         uint8_t index = getIndex(midimsg.channel, targetTrack, getBaseChannel(),
 119                                  getBaseTrack()) %
 120                         NUMBER_OF_BANKS;
 121         uint8_t data = midimsg.data1 & 0x0F;
 122         switch (data) {
 123             case 0xF: clearOverload(index); break;
 124             case 0xE: setOverload(index); break;
 125             case 0xD: break; // no meaning
 126             default: setValue(index, data); break;
 127         }
 128         display();
 129         return true;
 130     }
 131 
 132     void reset() override {
 133         for (uint8_t &value : values)
 134             value = 0;
 135     }
 136 
 137   private:
 138     void decayAll() override {
 139         for (uint8_t &value : values)
 140             if (getValueHelper(value) > 0)
 141                 value--;
 142     }
 143 
 144     uint8_t getRawValue() const override {
 145         return values[getBankSetting() % NUMBER_OF_BANKS];
 146     }
 147 
 148     inline bool matchTrack(uint8_t targetTrack) const {
 149         return BankableMIDIInputAddressable::matchAddress(
 150             targetTrack, getBaseTrack(), NUMBER_OF_BANKS);
 151     }
 152     inline bool matchChannel(uint8_t targetChannel) const override {
 153         return BankableMIDIInputAddressable::matchChannel(
 154             targetChannel, getBaseChannel(), NUMBER_OF_BANKS);
 155     }
 156     void setValue(uint8_t index, uint8_t newValue) {
 157         values[index] = setValueHelper(values[index], newValue);
 158     }
 159     void setOverload(uint8_t index) {
 160         values[index] = setOverloadHelper(values[index]);
 161     }
 162     void clearOverload(uint8_t index) {
 163         values[index] = clearOverloadHelper(values[index]);
 164     }
 165 
 166     uint8_t values[NUMBER_OF_BANKS] = {};
 167 };
