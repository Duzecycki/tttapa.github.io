 120 bool Selector::debounceButton(debouncedButton &button) {
 121     bool pressed = false;
 122     bool state = digitalRead(button.pin);
 123 
 124     int8_t stateChange = state - button.prevState;
 125 
 126     if (stateChange == falling) { // Button is pushed
 127         if (millis() - prevBounceTime > debounceTime) {
 128             pressed = true;
 129             prevBounceTime = millis();
 130         }
 131     }
 132     if (stateChange == rising) { // Button is released
 133         prevBounceTime = millis();
 134     }
 135     button.prevState = state;
 136     return pressed;
