   1 #pragma once
   2 
   3 #include <Selectors/Selector.h>
   4 #include <Banks/Bank.h>
   5 
   6 /**
   7  * @brief   A class for changing the setting of a Bank
   8  *          depending on the selection of a Selector.
   9  *
  10  * Can be used for selecting fader banks,
  11  * changing the channel of MIDI_Control_Element%s,
  12  * transposing notes, displaying different tracks
  13  * for MIDI_Input_Element%s, etc.
  14  *
  15  * Refer to the documentation of Selector for information
  16  * on the different modes of operation.
  17  */
  18 class BankSelector : public Selector {
  19   public:
  20     /**
  21      * @see Selector::Selector
  22      */
  23     template <class... Args>
  24     BankSelector(Bank &bank, Args... args) : Selector(args...), bank(bank) {}
  25 
  26     // Initializer_lists are not supported with variadic templates, so overload
  27     // them manually
  28 
  29     /**
  30      * @see Selector::Selector
  31      */
  32     // Multiple buttons, no LEDs
  33     template <class T, size_t N>
  34     BankSelector(Bank &bank, const T (&switchPins)[N])
  35         : Selector((const pin_t (&)[N])switchPins), bank(bank) {}
  36 
  37     /**
  38      * @see Selector::Selector
  39      */
  40     // One (+1), two (+1, -1) or multiple buttons, multiple LEDs
  41     template <class T, class S, size_t N, size_t M>
  42     BankSelector(Bank &bank, const T (&switchPins)[N], const S (&ledPins)[M])
  43         : Selector((const pin_t (&)[N])switchPins, (const pin_t (&)[M])ledPins),
  44           bank(bank) {}
  45 
  46     /**
  47      * @see Selector::Selector
  48      */
  49     // One (+1) or two (+1, -1) buttons, no LEDs
  50     template <class T, class S, size_t N>
  51     BankSelector(Bank &bank, const T (&switchPins)[N], S nb_settings)
  52         : Selector((const pin_t (&)[N])switchPins, (size_t)nb_settings),
  53           bank(bank) {}
  54 
  55   private:
  56     Bank &bank;
  57     /**
  58      * @brief   Update the bank setting with the new selector setting.
  59      */
  60     void updateImpl(uint8_t newSetting) { bank.setBankSetting(newSetting); }
  61 };
