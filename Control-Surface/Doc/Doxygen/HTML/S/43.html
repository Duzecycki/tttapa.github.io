  61 void Control_Surface_::updateMidiInput() {
  62     MIDI_read_t midiReadResult = MIDI().read();
  63 
  64     while (midiReadResult != NO_MESSAGE) {
  65         if (midiReadResult == CHANNEL_MESSAGE) {
  66             MIDI_message midichmsg = midi->getChannelMessage();
  67             MIDI_message_matcher midimsg(midichmsg);
  68 
  69 #ifdef DEBUG_MIDI_PACKETS
  70             if (midimsg.type != PROGRAM_CHANGE &&
  71                 midimsg.type != CHANNEL_PRESSURE)
  72                 DEBUG(">>> " << hex << +midichmsg.header << ' '
  73                              << +midimsg.data1 << ' ' << +midimsg.data2);
  74             else
  75                 DEBUG(">>> " << hex << +midichmsg.header << ' '
  76                              << +midimsg.data1);
  77 #endif
  78 
  79             if (midimsg.type == CC && midimsg.data1 == 0x79) {
  80                 // Reset All Controllers
  81                 DEBUG("Reset All Controllers");
  82                 MIDI_Input_CC::resetAll();
  83                 MIDI_Input_ChannelPressure::resetAll();
  84 
  85             } else if (midimsg.type == CC && midimsg.data1 == 0x7B) {
  86                 MIDI_Input_Note::resetAll();
  87             } else {
  88 
  89                 if (midimsg.type == CC) {
  90                     // Control Change
  91                     DEBUGFN(F("Updating CC elements with new MIDI message."));
  92                     MIDI_Input_CC::updateAllWith(midimsg);
  93 
  94                 } else if (midimsg.type == NOTE_OFF ||
  95                            midimsg.type == NOTE_ON) {
  96                     // Note
  97                     DEBUGFN(F("Updating Note elements with new MIDI message."));
  98                     MIDI_Input_Note::updateAllWith(midimsg);
  99 
 100                 } else if (midimsg.type == CHANNEL_PRESSURE) {
 101                     // Channel Pressure
 102                     DEBUGFN(F("Updating Channel Pressure elements with new "
 103                               "MIDI message."));
 104                     MIDI_Input_ChannelPressure::updateAllWith(midimsg);
 105                 }
 106             }
 107 
 108         } else if (midiReadResult == SYSEX_MESSAGE) {
 109             // System Exclusive
 110 #ifdef DEBUG_MIDI_PACKETS
 111             const uint8_t *data = midi->getSysExBuffer();
 112             size_t len = midi->getSysExLength();
 113             DEBUG_OUT << hex;
 114             for (size_t i = 0; i < len; i++)
 115                 DEBUG_OUT << data[i] << ' ';
 116             DEBUG_OUT << dec << endl;
 117 #endif
 118         }
 119         midiReadResult = midi->read();
 120     }
 121 }
