   5 /**
   6  * @brief   A class for applying hysteresis to a given input.
   7  *
   8  *          This reduces the noise by decreasing the resolution, and it prevents
   9  *          flipping back and forth between two values.
  10  *
  11  * <b>An example for `BITS` = 7 and an input from 0 to 1023</b>
  12  * ```
  13  *    7                                                     ┌───◄───┬───
  14  * o  6                                             ┌───◄───┼───►───┘
  15  * u  5                                     ┌───◄───┼───►───┘
  16  * t  4                             ┌───◄───┼───►───┘
  17  * p  3                     ┌───◄───┼───►───┘
  18  * u  2             ┌───◄───┼───►───┘
  19  * t  1     ┌───◄───┼───►───┘
  20  *    0 ────┴───►───┘
  21  *      0      128     256     384     512     640     768     896    1023
  22  *                                i n p u t
  23  * ```
  24  * 
  25  * @tparam  BITS
  26  *          The number of bits to decrease in resolution.  
  27  *          Increasing this number will result in a decrease in fluctuations.
  28  */
  29 template <uint8_t BITS>
  30 class Hysteresis {
  31   public:
  32     /**
  33      * @brief     Get the output level for a given input.
  34      *
  35      * @param     inputLevel

  39     uint16_t getOutputLevel(uint16_t inputLevel);

  41   private:

  49 template <uint8_t BITS>
  50 inline uint16_t Hysteresis<BITS>::getOutputLevel(uint16_t inputLevel) {
  51     uint16_t previousLevelFull = ((uint16_t)previousLevel << BITS) | offset;
  52     uint16_t lowerbound = previousLevelFull;
  53     if (previousLevel > 0)
  54         lowerbound -= margin;
  55     uint16_t upperbound = previousLevelFull + margin;
  56     if (inputLevel < lowerbound || inputLevel > upperbound)
  57         previousLevel = inputLevel >> BITS;
  58     return previousLevel;
