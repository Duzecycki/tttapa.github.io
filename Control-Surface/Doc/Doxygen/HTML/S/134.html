   7 /**
   8  * @brief   The abstract class that all MIDI elements (both inputs and outputs)
   9  *          inherit from.  
  10  *          It declares methods for updating these elements, resetting them,
  11  *          and changing settings related to banks (channel/address offset
  12  *          and number of tracks per bank).
  13  */
  14 class MIDI_Element {
  15     friend class Control_Surface_;
  16 
  17   protected:
  18     /**
  19      * @brief Reset to default value (currently only used for VU meters)
  20      */
  21     virtual void reset() {}
  22     /**
  23      * @brief Check if the state or value of the MIDI Element has changed since
  24      *        last update, if so, send a MIDI event or update
  25      *        the MIDI_Input_Element (write buffer to output etc.)
  26      */
  27     virtual void update() = 0;
  28 
  29   public:
  30     /**
  31      * @see FilteredAnalog::map
  32      */
  33     virtual void map(int (*fn)(int)) {}
  34 
  35     /**
  36      * @brief Set the channel offset.
  37      *
  38      * @param offset
  39      */
  40     virtual void setChannelOffset(uint8_t offset) { channelOffset = offset; }
  41     /**
  42      * @brief Set the address (note or controller number) offset
  43      *
  44      * @param offset
  45      */
  46     virtual void setAddressOffset(uint8_t offset) { addressOffset = offset; }
  47     /**
  48      * @brief Set the number of tracks per bank
  49      *
  50      * @param tracks
  51      */
  52     virtual void settracksPerBank(uint8_t tracks) { tracksPerBank = tracks; }
  53 
  54   protected:
  55     uint8_t channelOffset = 0;
  56     uint8_t addressOffset = 0;
  57     uint8_t tracksPerBank = 1;
