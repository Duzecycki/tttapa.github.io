   6 /**
   7  * @brief   A class for serial-in/parallel-out shift registers, 
   8  *          like the 74HC595. 
   9  */
  10 class ShiftRegisterOut : public ExtendedIOElement {
  11   public:
  12     /**
  13      * @brief   Create a new ShiftRegisterOut object with a shift register
  14      *          connected to the given pins, with a given bit order,
  15      *          and a given number of outputs.
  16      * 
  17      * Multiple shift registers can be cascaded:
  18      * ```
  19      * clockPin >──────────────┬─────────────────────────────┬─────────────────────────────┐            
  20      *              ┎━━━━━━━━━━┷━━━━━━━━━━━┓      ┎━━━━━━━━━━┷━━━━━━━━━━━┓      ┎━━━━━━━━━━┷━━━━━━━━━━━┓
  21      *              ┃        SH_CP         ┃      ┃        SH_CP         ┃      ┃        SH_CP         ┃
  22      * dataPin  >───┨ Data in     Data out ┠──────┨ Data in     Data out ┠──────┨ Data in     Data out ┃
  23      *              ┃        ST_CP         ┃      ┃        ST_CP         ┃      ┃        ST_CP         ┃
  24      *              ┗━━━━━━━━━━┯━━━━━━━━━━━┛      ┗━━━━━━━━━━┯━━━━━━━━━━━┛      ┗━━━━━━━━━━┯━━━━━━━━━━━┛
  25      * latchPin >──────────────┴─────────────────────────────┴─────────────────────────────┘            
  26      * ```
  27      * 
  28      * @param   dataPin
  29      *          The digital output pin connected to the serial data input (DS)
  30      *          of the shift register.
  31      * @param   clockPin
  32      *          The digital output pin connected to the clock input (SH_CP)
  33      *          of the shift register.
  34      * @param   latchPin
  35      *          The digital output pin connected to the latch pin (ST_CP)
  36      *          of the shift register.
  37      */
  38     ShiftRegisterOut(pin_t dataPin, pin_t clockPin, pin_t latchPin,
  39                      uint8_t bitOrder, pin_t length = 8);
  40 
  41     /**
  42      * @brief   The pinMode function is not implemented because the mode is

  46     }
  47 
  48     /**

  51      * @param   pin
  52      *          The shift register pin to set.

  59     /**

  62      * @param   pin

  67     /**
  68      * @brief   The analogRead function is deprecated because a shift

  75     /**
  76      * @brief   The analogWrite function is not deprecated because a shift

  91     /**
