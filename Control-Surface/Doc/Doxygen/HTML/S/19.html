   8 /**
   9  * @brief   A class for MIDI interfaces sending and receiving MIDI messages
  10  *          over a Stream.
  11  */
  12 class StreamMIDI_Interface : public MIDI_Interface {
  13   public:
  14     /**
  15      * @brief   Construct a StreamMIDI_Interface on the given Stream.
  16      * 
  17      * @param   stream
  18      *          The Stream interface.
  19      */
  20     StreamMIDI_Interface(Stream &stream)
  21         : MIDI_Interface(parser), stream(stream) {}
  22 
  23     virtual MIDI_read_t read() override {
  24         while (stream.available() > 0) {
  25             uint8_t midiByte = stream.read();
  26             MIDI_read_t parseResult = parser.parse(midiByte);
  27             if (parseResult != NO_MESSAGE)
  28                 return parseResult;
  29         }
  30         return NO_MESSAGE;
  31     }
  32 
  33   protected:
  34     SerialMIDI_Parser parser;
  35 
  36     virtual void sendImpl(uint8_t m, uint8_t c, uint8_t d1,
  37                           uint8_t d2) override {
  38         stream.write(m | c); // Send the MIDI message over the stream
  39         stream.write(d1);
  40         stream.write(d2);
  41         stream.flush(); // TODO
  42     }
  43 
  44     virtual void sendImpl(uint8_t m, uint8_t c, uint8_t d1) override {
  45         stream.write(m | c); // Send the MIDI message over the stream
  46         stream.write(d1);
  47         stream.flush(); // TODO
  48     }
  49 
  50   protected:
  51     Stream &stream;
  52 };

  54 /**
  55  * @brief   A wrapper class for MIDI interfaces sending and receiving 
  56  *          MIDI messages over a Serial port of class T.
  57  * 
  58  * @note    This is a template class because the class of the Serial object
  59  *          is completely different on different architectures, and they
  60  *          do not share a common super-class that has a `begin` method.
  61  */
  62 template <typename T>
  63 class SerialMIDI_Interface : public StreamMIDI_Interface {
  64   public:
  65     /**
  66      * @brief   Create a new MIDI Interface on the given Serial interface
  67      *          with the given baud rate.
  68      * 
  69      * @param   serial
  70      *          The Serial interface.
  71      * @param   baud
  72      *          The baud rate for the Serial interface.
  73      */
  74     SerialMIDI_Interface(T &serial, unsigned long baud)
  75         : StreamMIDI_Interface(serial), serial(serial), baud(baud) {}
  76     /**
  77      * @brief   Start the Serial interface at the predefined baud rate.
  78      */
  79     virtual void begin() override { serial.begin(baud); }
  80 
  81   private:
  82     T &serial;
  83     const unsigned long baud;
  84 };
