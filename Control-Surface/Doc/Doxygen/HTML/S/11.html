  23      * @brief   Update a character.
  24      */
  25     virtual bool updateImpl(const MIDI_message_matcher &midimsg) {
  26         uint8_t targetAddress = midimsg.data1;
  27         uint8_t index = targetAddress - getBaseAddress();
  28         // decimal point → 0x80, no decimal point → 0x00
  29         uint8_t decimalPt = (midimsg.data2 & 0x40) << 1;
  30         uint8_t data2 = midimsg.data2 & 0x3F;
  31         uint8_t character = data2 >= 0x20 ? data2 : data2 + 0x40;
  32         DEBUGREF(F("\tCharacter @") << getBaseAddress() << ':' << index << '\t'
  33                                     << character << (decimalPt ? '.' : ' '));
  34         character |= decimalPt;
  35         text[index] = character;
  36         return true;
  37     }
  38 
  39     /**
  40      * @brief   Check if the address of the incoming MIDI message
  41      *          matches an address of this element.
  42      */
  43     virtual inline bool matchAddress(uint8_t targetAddress) const {
  44         return getBaseAddress() <= targetAddress &&
  45                targetAddress < getBaseAddress() + LENGTH;
  46     }

  49     /**
  50      * @brief   Copy the ASCII text into the given buffer.
  51      *
  52      * @param   buffer
