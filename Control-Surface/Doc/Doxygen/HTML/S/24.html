   8 /**
   9  * @brief   Append a node to a linked list.
  10  * 
  11  * @param   node
  12  *          A pointer to the node to be appended.
  13  * @param   first
  14  *          The pointer to the first node of the linked list.
  15  * @param   last
  16  *          The pointer to the last node of the linked list.
  17  */
  18 template <class Node>
  19 void append(Node *node, Node *&first, Node *&last) {
  20     if (first == nullptr)
  21         first = node;
  22     node->previous = last;
  23     if (node->previous != nullptr)
  24         node->previous->next = node;
  25     last = node;
  26     node->next = nullptr;
  27 }

  29 /**
  30  * @brief   Remove a node from a linked list.
  31  * 
  32  * @param   node
  33  *          A pointer to the node to be removed.
  34  * @param   first
  35  *          The pointer to the first node of the linked list.
  36  * @param   last
  37  *          The pointer to the last node of the linked list.
  38  */
  39 template <class Node>
  40 void remove(Node *node, Node *&first, Node *&last) {
  41     if (node->previous != nullptr)
  42         node->previous->next = node->next;
  43     if (node == last)
  44         last = node->previous;
  45     if (node->next != nullptr)
  46         node->next->previous = node->previous;
  47     if (node == first)
  48         first = node->next;
  49 }

  51 /**
  52  * @brief   Move down the given node in a linked list.
  53  * 
  54  * For example: moving down node `C`:
  55  * ```
  56  * Before:  ... → A → B → C → D → ...
  57  * After:   ... → A → C → B → D → ...
  58  * ```
  59  * @param   nodeC
  60  *          A pointer to the node to be moved down.
  61  * @param   first
  62  *          The pointer to the first node of the linked list.
  63  * @param   last
  64  *          The pointer to the last node of the linked list.
  65  */
  66 template <class Node>
  67 void moveDown(Node *nodeC, Node *&first, Node *&last) {
  68     Node *nodeB = nodeC->previous;
  69     if (nodeB == nullptr) // Can't move first node further down
  70         return;
  71     Node *nodeA = nodeB->previous;
  72     Node *nodeD = nodeC->next;
  73 
  74     if (nodeA != nullptr)
  75         nodeA->next = nodeC;
  76     else
  77         first = nodeC;
  78     nodeB->next = nodeD;
  79     nodeB->previous = nodeC;
  80     nodeC->next = nodeB;
  81     nodeC->previous = nodeA;
  82     if (nodeD != nullptr)
  83         nodeD->previous = nodeB;
  84     else
  85         last = nodeB;
  86 }
