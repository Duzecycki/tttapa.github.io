   5 /**
   6  * @brief   A class for single-pole infinite impulse response filters
   7  *          or exponential moving average filters.
   8  *
   9  * Straightforward implementation of the difference equation with a slight
  10  * optimization by using a factor of two as the pole location (this means
  11  * that no division or floating point operations are required).
  12  *
  13  * Difference equation: @f$ y[n] = \alpha·x[n]+(1-\alpha)·y[n-1] @f$
  14  * where @f$ \alpha = \left(\frac{1}{2}\right)^{K} @f$, @f$ x @f$ is the
  15  * input sequence, and @f$ y @f$ is the output sequence.
  16  *
  17  * [An in-depth explanation of the EMA filter]
  18  * (https://tttapa.github.io/Pages/Mathematics/Systems-and-Control-Theory/Digital-filters/Exponential%20Moving%20Average/)
  19  * 
  20  * Fixed-point integer arithmetic with a precision of @f$ 2K @f$ is used.
  21  *
  22  * @tparam  K
  23  *          The amount of bits to shift by. This determines the location
  24  *          of the pole in the EMA transfer function, and therefore the
  25  *          cut-off frequency.
  26  * @tparam  int_t
  27  *          The (signed) integer type to use for the input, intermediate values
  28  *          and the output.
  29  *          Should be at least @f$ M+1+2K @f$ bits wide, where @f$ M @f$
  30  *          is the maximum number of bits of the input.
  31  *          In case of the Arduino's built-in ADC,
  32  *          @f$ M = 10 = \log_2(1024) @f$.
  33  *
  34  */
  35 template <uint8_t K, class int_t>
  36 class EMA {
  37   public:
  38     /**
  39      * @brief   Filter the input: Given @f$ x[n] @f$, calculate @f$ y[n] @f$.
  40      *
  41      * @param   input

  45     int_t filter(int_t input);
