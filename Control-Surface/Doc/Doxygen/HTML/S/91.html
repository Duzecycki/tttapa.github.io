  12 class AnalogMultiplex : public ExtendedIOElement {

  14     /**
  15      * @brief   Create a new AnalogMultiplex object on the given pins.
  16      * 
  17      * @param   analogPin
  18      *          The analog input pin connected to the output of the multiplexer.
  19      * @param   addressPins
  20      *          A list of the pins connected to the address lines of the
  21      *          multiplexer.
  22      * @note    The list of address pins is copied. This means that it is safe

  25     template <class P, size_t N>

  28         this->addressPins = new pin_t[N];

  30         begin();
  31     }
  32     ~AnalogMultiplex() { delete[] addressPins; }

  38      * 
  39      * @note    This applies to all pins of this multiplexer.  
  40      *          It is not possible to enable the pull-up resistor on some of
  41      *          the inputs and not on others.  
  42      * 
  43      * @param   (unused)
  44      * @param   mode
  45      *          The new mode of the analog input pin:
