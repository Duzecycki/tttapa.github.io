79  // If the Channel Message callback exists, call it to see if we have to
80  // continue handling it.
81  if (channelMessageCallback && channelMessageCallback(midichmsg))
82  return;
83 
84  if (midimsg.type == CC && midimsg.data1 == 0x79) {
85  // Reset All Controllers
86  DEBUG(F("Reset All Controllers"));
87  MIDIInputElementCC::resetAll();
88  MIDIInputElementChannelPressure::resetAll();
89  } else if (midimsg.type == CC && midimsg.data1 == MIDI_CC::All_Notes_Off) {
90  MIDIInputElementNote::resetAll();

92  if (midimsg.type == CC) {

94  DEBUGFN(F("Updating CC elements with new MIDI message."));
95  MIDIInputElementCC::updateAllWith(midimsg);
96 
97  } else if (midimsg.type == NOTE_OFF || midimsg.type == NOTE_ON) {
98  // Note
99  DEBUGFN(F("Updating Note elements with new MIDI message."));
100  MIDIInputElementNote::updateAllWith(midimsg);
101 
102  } else if (midimsg.type == CHANNEL_PRESSURE) {
103  // Channel Pressure
104  DEBUGFN(F("Updating Channel Pressure elements with new "
105  "MIDI message."));
106  MIDIInputElementChannelPressure::updateAllWith(midimsg);

108  // Channel Pressure
