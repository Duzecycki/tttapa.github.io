228  SimulationResult simulate(DiscreteController<Nx, Nu, Ny> &controller,
229  ReferenceFunction &r, VecX_t x_start,
230  const AdaptiveODEOptions &opt) override {
231  SimulationResult result = {};
232  double Ts = controller.Ts;
233  size_t N = floor((opt.t_end - opt.t_start) / Ts) + 1;
234  VecX_t curr_x = x_start;
235  AdaptiveODEOptions curr_opt = opt;
236  for (size_t i = 0; i < N; ++i) {
237  double t = opt.t_start + Ts * i;
238  curr_opt.t_start = t;
239  curr_opt.t_end = t + Ts;
240  VecR_t curr_ref = r(t);
241  VecU_t curr_u = controller(curr_x, curr_ref);
242  result.resultCode |= simulate(std::back_inserter(result.time),
243  std::back_inserter(result.solution),
244  curr_u, curr_x, curr_opt);
245  curr_x = result.solution.back();
246  result.time.pop_back();
247  result.solution.pop_back();
248  }
249  return result;
250  }

252  ObserverSimulationResult
253  simulate(DiscreteController<Nx, Nu, Ny> &controller,
254  DiscreteObserver<Nx, Nu, Ny> &observer,
255  TimeFunctionT<VecU_t> &randFnW, TimeFunctionT<VecY_t> &randFnV,
256  ReferenceFunction &r, VecX_t x_start,
257  const AdaptiveODEOptions &opt) override {
258  assert(controller.Ts == observer.Ts);
259  ObserverSimulationResult result = {};
260  double Ts = controller.Ts;
261  size_t N = numberOfSamplesInTimeRange(opt.t_start, Ts, opt.t_end);
262 
263  // pre-allocate memory for result vectors
264  result.sampledTime.reserve(N);
265  result.estimatedSolution.reserve(N);
266  result.control.reserve(N);
267  result.output.reserve(N);
268 
269  // actual state = inital state
270  VecX_t curr_x = x_start;
271  // estimated state = inital state
272  VecX_t curr_x_hat = x_start;
273  AdaptiveODEOptions curr_opt = opt;
274  // For each time step
275  for (size_t k = 0; k < N; ++k) {
276  // current time, and integration range
277  double t = opt.t_start + Ts * k;
278  curr_opt.t_start = t;
279  curr_opt.t_end = t + Ts;
280  // disturbance w and sensor noise v
281  auto w = randFnW(t);
282  auto v = randFnV(t);
283  // reference signal
284  VecR_t curr_ref = r(t);
285  // calculate the control signal, based on current estimated state
286  // and current reference
287  VecU_t curr_u = controller(curr_x_hat, curr_ref);
288  // the output of the real system is the output of the system, given
289  // the actual state and the control signal, plus the sensor
290  // noise
291  VecY_t clean_y = this->getOutput(curr_x, curr_u); // no noise
292  VecY_t y = quaternionStatesAdd(clean_y, v); // add noise
293  // add the time, estimate state, control signal and output to the
294  // result vectors
295  result.sampledTime.push_back(t);
296  result.estimatedSolution.push_back(curr_x_hat);
297  result.control.push_back(curr_u);
298  result.output.push_back(y);
299 
300  // calculate the estimated state for the next time step
301  // k+1 k k k
302  curr_x_hat = observer.getStateChange(curr_x_hat, y, curr_u);
303 
304  // simulate the continuous system over this time step [t, t + Ts]
305  // and add the time points and states to the result
306  result.resultCode |= simulate(std::back_inserter(result.time),
307  std::back_inserter(result.solution),
308  curr_u + w, curr_x, curr_opt);
309  // update the actual state using the result of the continuous
310  // simulation at t + Ts
311  curr_x = result.solution.back();
312  // start and end point are included by `simulate`, so remove doubles
313  result.time.pop_back();
314  result.solution.pop_back();
315  }
316  return result;
317  }
