284  ControllerSimulationResult
285  simulate(DiscreteController<Nx, Nu, Ny> &controller, ReferenceFunction &r,
286  VecX_t x_start, const AdaptiveODEOptions &opt) {
287  ControllerSimulationResult result = {};
288  double Ts = controller.Ts;
289  size_t N = numberOfSamplesInTimeRange(opt.t_start, Ts, opt.t_end);
290  // pre-allocate memory for result vectors
291  result.sampledTime.reserve(N);
292  result.control.reserve(N);
293  result.reference.reserve(N);
294 
295  // actual state = inital state
296  VecX_t curr_x = x_start;
297  AdaptiveODEOptions curr_opt = opt;
298  // For each time step
299  for (size_t i = 0; i < N; ++i) {
300  // current time, and integration range
301  double t = opt.t_start + Ts * i;
302  curr_opt.t_start = t;
303  curr_opt.t_end = t + Ts;
304  curr_opt.maxiter = opt.maxiter - result.iterations;
305  // reference signal
306  VecR_t curr_ref = r(t);
307  // calculate the control signal, based on current state
308  // and current reference
309  VecU_t curr_u = controller(curr_x, curr_ref);
310  // add the time, control signal and reference to the
311  // result vectors
312  result.sampledTime.push_back(t);
313  result.control.push_back(curr_u);
314  result.reference.push_back(curr_ref);
315  // simulate the continuous system over this time step [t, t + Ts]
316  // and add the time points and states to the result
317  auto curr_result = this->simulate(
318  std::back_inserter(result.time),
319  std::back_inserter(result.solution), curr_u, curr_x, curr_opt);
320  result.resultCode |= curr_result.first;
321  result.iterations += curr_result.second;
322  // update the actual state using the result of the continuous
323  // simulation at t + Ts
324  curr_x = result.solution.back();
325  // start and end point are included by `simulate`, so remove doubles
326  result.time.pop_back();
327  result.solution.pop_back();
328  }
329  return result;
330  }

333  ODEResultCode simulateRealTime(DiscreteController<Nx, Nu, Ny> &controller,
334  ReferenceFunction &r, VecX_t x_start,
335  const AdaptiveODEOptions &opt, F &callback) {
336  ODEResultCode resultCode;
337  double Ts = controller.Ts;
338  size_t N = numberOfSamplesInTimeRange(opt.t_start, Ts, opt.t_end);
339  VecX_t curr_x = x_start;
340  AdaptiveODEOptions curr_opt = opt;
341  for (size_t i = 0; i < N; ++i) {
342  double t = opt.t_start + Ts * i;
343  curr_opt.t_start = t;
344  curr_opt.t_end = t + Ts;
345  VecR_t curr_ref = r(t);
346  VecU_t curr_u = controller(curr_x, curr_ref);
347  if (!callback(t, curr_x, curr_u))
348  break;
349  auto result = this->simulateEndResult(curr_u, curr_x, curr_opt);
350  curr_opt.maxiter -= result.iterations;
351  resultCode |= result.resultCode;
352  if (resultCode & ODEResultCodes::MAXIMUM_ITERATIONS_EXCEEDED)
353  break;
354  curr_x = result.solution[0];
355  }
356  return resultCode;
