373  SimulationResult simulate(DiscreteController<Nx, Nu, Ny> &controller,
374  ReferenceFunction &r, VecX_t x_start,
375  const AdaptiveODEOptions &opt) override {
376  SimulationResult result = {};
377  double Ts = controller.Ts;
378  size_t N = floor((opt.t_end - opt.t_start) / Ts) + 1;
379  VecX_t curr_x = x_start;
380  AdaptiveODEOptions curr_opt = opt;
381  for (size_t i = 0; i < N; ++i) {
382  double t = opt.t_start + Ts * i;
383  curr_opt.t_start = t;
384  curr_opt.t_end = t + Ts;
385  VecR_t curr_ref = r(t);
386  VecU_t curr_u = controller(curr_x, curr_ref);
387  result.resultCode |= this->simulate(
388  std::back_inserter(result.time),
389  std::back_inserter(result.solution), curr_u, curr_x, curr_opt);
390  curr_x = result.solution.back();
391  result.time.pop_back();
392  result.solution.pop_back();
393  }
394  return result;

431  ObserverSimulationResult
432  simulate(DiscreteController<Nx, Nu, Ny> &controller,
433  DiscreteObserver<Nx, Nu, Ny> &observer,
434  TimeFunctionT<VecU_t> &randFnW, TimeFunctionT<VecY_t> &randFnV,
435  ReferenceFunction &r, VecX_t x_start,
436  const AdaptiveODEOptions &opt) override {
437  assert(controller.Ts == observer.Ts);
438  ObserverSimulationResult result = {};
439  double Ts = controller.Ts;
440  size_t N = numberOfSamplesInTimeRange(opt.t_start, Ts, opt.t_end);
441 
442  // pre-allocate memory for result vectors
443  result.sampledTime.reserve(N);
444  result.estimatedSolution.reserve(N);
445  result.control.reserve(N);
446  result.output.reserve(N);
447 
448  // actual state = inital state
449  VecX_t curr_x = x_start;
450  // estimated state = inital state
451  VecX_t curr_x_hat = x_start;
452  AdaptiveODEOptions curr_opt = opt;
453  // For each time step
454  for (size_t k = 0; k < N; ++k) {
455  // current time, and integration range
456  double t = opt.t_start + Ts * k;
457  curr_opt.t_start = t;
458  curr_opt.t_end = t + Ts;
459  // disturbance w and sensor noise v
460  auto w = randFnW(t);
461  auto v = randFnV(t);
462  // reference signal
463  VecR_t curr_ref = r(t);
464  // calculate the control signal, based on current estimated state
465  // and current reference
466  VecU_t curr_u = controller(curr_x_hat, curr_ref);
467  // the output of the real system is the output of the system, given
468  // the actual state and the control signal, plus the sensor
469  // noise
470  VecY_t clean_y = this->getOutput(curr_x, curr_u); // no noise
471  VecY_t y = quaternionStatesAdd(clean_y, v); // add noise
472  // add the time, estimate state, control signal and output to the
473  // result vectors
474  result.sampledTime.push_back(t);
475  result.estimatedSolution.push_back(curr_x_hat);
476  result.control.push_back(curr_u);
477  result.output.push_back(y);
478 
479  // calculate the estimated state for the next time step
480  // k+1 k k k
481  curr_x_hat = observer.getStateChange(curr_x_hat, y, curr_u);
482 
483  // simulate the continuous system over this time step [t, t + Ts]
484  // and add the time points and states to the result
485  result.resultCode |=
486  this->simulate(std::back_inserter(result.time),
487  std::back_inserter(result.solution), curr_u + w,
488  curr_x, curr_opt);
489  // update the actual state using the result of the continuous
490  // simulation at t + Ts
491  curr_x = result.solution.back();
492  // start and end point are included by `simulate`, so remove doubles
493  result.time.pop_back();
494  result.solution.pop_back();
495  }
496  return result;
