131  HoughResult findLineAngleAccurateRange(Pixel px,
132  angle_t centerAngle) const {
133  static_assert(2 * N < angle_t::resolution());
134  std::array<HoughResult, 2 * N + 1> houghRes;
135  uint centerAngleIndex = centerAngle.getIndex();
136 
137  // If the range contains no discontinuities
138  // ├───┬───┬───╤═══╤═══╤═══╪═══╤═══╤═══╤───┬───┬───┤
139  // 0 C-N C C+N R-1
140  // C = center
141  // N = number of steps around the center
142  // R = resolution (R steps = 720°)
143  if (centerAngleIndex - N < angle_t::resolution() &&
144  centerAngleIndex + N < angle_t::resolution()) {
145  uint initialAngleIndex = centerAngleIndex - N;
146  for (uint i = 0; i <= 2 * N; ++i)
147  houghRes.at(i) = hough(px, angle_t(initialAngleIndex + i));
148  }
149  // If the 0-2π discontinuity is in the first part of the range
150  // (< centerAngle)
151  // ╞═══╪═══╤═══╤═══╤───┬───┬───┬───┬───┬───┬───╤═══╡
152  // 0 C C+N C-N R-1
153  else if (centerAngleIndex + N < angle_t::resolution()) {
154  uint initialAngleIndex =
155  angle_t::resolution() + centerAngleIndex - N;
156  uint i = 0;
157  for (uint j = initialAngleIndex; j < angle_t::resolution();
158  ++i, ++j)
159  houghRes.at(i) = hough(px, angle_t(j));
160  for (uint j = 0; j <= centerAngleIndex + N; ++i, ++j)
161  houghRes.at(i) = hough(px, angle_t(j));
162  }
163  // If the 0-2π discontinuity is in the second part of the range
164  // (> centerAngle)
165  // ╞═══╤───┬───┬───┬───┬───┬───┬───╤═══╤═══╤═══╪═══╡
166  // 0 C+N C-N C R-1
167  else if (angle_t::resolution() - N < angle_t::resolution()) {
168  uint initialAngleIndex = centerAngleIndex - N;
169  uint i = 0;
170  for (uint j = initialAngleIndex; j < angle_t::resolution();
171  ++i, ++j)
172  houghRes.at(i) = hough(px, angle_t(j));
173  for (uint j = 0; j <= N + centerAngleIndex - angle_t::resolution();
174  ++i, ++j)
175  houghRes.at(i) = hough(px, angle_t(j));
176  } else {
177  throw std::runtime_error("Something is very wrong");
178  }
179  // If there are multiple elements with the same maximum count,
180  // max_element will return an iterator to the first element with this
181  // count
182  auto max = std::max_element(houghRes.begin(), houghRes.end());
183 
184  // Find the first element after the first maximum that has a lower count
185  // than ⅞ of the maximum count
186  // and then subtract one to get the last element that has the maximum
187  // count.
188  // If no element with a lower count is found, find_if returns
189  // houghRes.end(), which means that last_max will be an iterator to
190  // the last element of the array.
191  // In that case, there's probably something wrong: the line might be
192  // too thick, and all counts within the specified range are very high.
193 
194  // predicate res.count < 7 * max->count / 8;
195  auto predicateLessThanSevenEighths = [&](auto res) {
196  return res.count < max->count - max->count / 8;
197  };
198  // std::find_if returns an iterator the first element that satisfies the
199  // predicate
200  //
201  auto first_max =
202  std::find_if(std::make_reverse_iterator(max), //
203  houghRes.rend(), predicateLessThanSevenEighths);
204  auto last_max =
205  std::find_if(max, //
206  houghRes.end(), predicateLessThanSevenEighths);
207 
208  if (first_max == houghRes.rend() || last_max == houghRes.end())
209  cerr << ANSIColors::red
210  << "Warning: angle with maximum count could lie outside of "
211  "the specified range\r\n"
212  << ANSIColors::reset;
213 
214  if (first_max != houghRes.rbegin())
215  --first_max;
216  if (last_max != houghRes.begin())
217  --last_max;
218 
219  return {
220  angle_t::average(first_max->angle, last_max->angle),
221  first_max->count,
222  };

300  optional<GetMiddleResult> getMiddle(Pixel pointOnLine, CosSin lineAngle,
301  uint maxLineGap = MAX_GAP) const {
302  // If the given point does not lie on a line, we can't find the middle
303  // of the line
304  if (get(pointOnLine) == 0x00)
305  return std::nullopt;
306 
307  CosSin oppositeAngle = lineAngle.opposite();
308  // Get the half width of the line in four directions.
309  // U2 U1
310  // ↑↑↑ ↑↑↑
311  // ---o---> line
312  // ↓↓↓ ↓↓↓
313  // L2 L1
314  uint widthUpper1 = getWidthAtPointOnLine(pointOnLine, lineAngle, //
315  maxLineGap / 2, true);
316  uint widthLower1 = getWidthAtPointOnLine(pointOnLine, lineAngle, //
317  maxLineGap / 2, false);
318  uint widthUpper2 = getWidthAtPointOnLine(pointOnLine, oppositeAngle, //
319  maxLineGap / 2, false);
320  uint widthLower2 = getWidthAtPointOnLine(pointOnLine, oppositeAngle, //
321  maxLineGap / 2, true);
322 
323  // Use the maximum of the upper width and the maximum of the lower width
324  uint widthUpper = std::max(widthUpper1, widthUpper2);
325  uint widthLower = std::max(widthLower1, widthLower2);
326 
327  // If either of the widths exceeds the MAX_LINE_WIDTH, the result is
328  // invalid: it is probably a large white spot, not a thin line
329  if (widthUpper >= MAX_LINE_WIDTH || widthLower >= MAX_LINE_WIDTH)
330  return std::nullopt;
331 
332  // Use the difference between the two widths to move the given point
333  // towards the center
334  int middlePointCorrection_x2 = widthUpper - widthLower;
335  uint middlePointCorrDistance = std::abs(middlePointCorrection_x2) / 2;
336  // Should we move up or down? (+90° or -90° from the estimated angle)
337  bool corrDirection = middlePointCorrection_x2 > 0;
338  CosSin corrAngle = lineAngle.perpendicular(corrDirection);
339  Pixel middle = move(pointOnLine, corrAngle, middlePointCorrDistance);
340  return GetMiddleResult{middle, widthUpper + widthLower - 1};
341  }

705  optional<LineResult> findNextLine(LineResult line, bool direction,
706  uint minDistance = 0,
707  uint offset = 0) const {
708  angle_t angle = line.angle;
709  angle_t perp = angle.perpendicular(direction);
710 
711  Pixel searchStart = line.lineCenter;
712  searchStart = move(searchStart, perp, 2 * line.width + offset);
713  if (minDistance)
714  searchStart = move(searchStart, line.angle, minDistance);
715 
716  uint minWidth = line.width / 3;
717 
718  BresenhamLine path = {searchStart, angle, W, H};
719  Pixel pixel;
720  do {
721  // Follow the search path
722  pixel = path.next();
723  // Find the first white pixel
724  while (path.hasNext() && get(pixel) == 0x00)
725  pixel = path.next();
726  Pixel firstWhite = pixel;
727  // Find the first black pixel
728  while (path.hasNext() && get(pixel) != 0x00)
729  pixel = path.next();
730  Pixel firstBlack = pixel; // TODO: OBOE, but I don't really care
731  // In the center of this range of white pixels, try to find the
732  // perpendicular line
733  Pixel middle = Pixel::average(firstBlack, firstWhite);
734  optional<LineResult> possibleLine =
735  checkLine(middle, perp, minWidth);
736  // If a line has been found, return it
737  if (possibleLine)
738  return possibleLine;
739  // Otherwise, keep looking along the search path
740  } while (path.hasNext());
741  // If we reach the end of the frame without finding a perpendicular line
742  // return no result, so the calling function can call again with a
743  // different offset (or give up)
744  return std::nullopt;

794  Square findSquare() {
795  Square sq;
796 
797  try {
798  // Get the line closest to the center of the frame.
799  // the result is an array of two half-lines, approximately opposite
800  // of each other
801  auto firstLines = getFirstTwoLines();
802  sq.lines[0] = firstLines[0];
803  sq.lines[1] = firstLines[1];
804 
805  // Determine the direction to turn in (+90° or -90°)
806  // Check if the first line lies left or right of the center of the frame
807  // and pick the direction that will result in the square containing the
808  // center point (if possible)
809  bool direction = false;
810  if (sq.lines[0].has_value()) {
811  Line mathLine = {sq.lines[0]->lineCenter, sq.lines[0]->angle};
812  direction = mathLine.leftOfPoint(center());
813  }
814 
815  // TODO: maybe try again once if it fails?
816  sq.lines[2] = findNextLine(sq.lines[0], direction); // second line
817  sq.lines[3] = findNextLine(sq.lines[1], !direction); // third line
818 
819  // If we found all lines so far (first two, second, and third)
820  if (sq.lines[2].has_value() && sq.lines[3].has_value()) {
821  sq.points[0] = intersect(*sq.lines[0], *sq.lines[2]);
822  sq.points[1] = intersect(*sq.lines[1], *sq.lines[3]);
823 
824  // Calculate the distance between the first two points
825  // (note that this is not the Euclidian distance, but the
826  // number of pixels that would be drawn by Bresenham)
827  uint minDistance = std::floor(
828  std::max(std::abs(sq.points[0]->x - sq.points[1]->x),
829  std::abs(sq.points[0]->y - sq.points[1]->y)));
830  // Use 3/4 of this distance as a minimum distance for where to
831  // start looking for the fourth line, as it's very likely that
832  // the other sides of the square are roughly the same length as
833  // the first side
834  minDistance -= minDistance / 4;
